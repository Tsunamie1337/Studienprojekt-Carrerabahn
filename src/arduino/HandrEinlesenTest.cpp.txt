#include <Arduino.h>

// ── Pins ─────────────────────────────────────────────────────

const int PIN_GAS    = 3;   // ADC – Gelb
const int PIN_TASTER = 5;   // Digital – Grün

// ── Standardwerte (aus deiner Messung) ───────────────────────
int adc_kein_gas = 3;      // Trigger losgelassen
int adc_vollgas  = 2027;   // Trigger voll gedrueckt

// ── Kalibrierungsdauer ────────────────────────────────────────
const int   KALI_DAUER_MS = 5000;
const float DEADZONE      = 0.03f;

// ── Taster-Entprellung ────────────────────────────────────────
const int     DEBOUNCE_MS     = 20;
bool          letzter_zustand = HIGH;
unsigned long letzter_wechsel = 0;

// ─────────────────────────────────────────────────────────────
String serial_lesen() {
  String input = "";
  while (Serial.available()) {
    char c = Serial.read();
    if (c == '\n' || c == '\r') continue;
    input += c;
  }
  input.trim();
  input.toLowerCase();
  return input;
}

String warte_auf_eingabe() {
  while (true) {
    delay(50);
    if (Serial.available()) {
      String s = serial_lesen();
      if (s.length() > 0) return s;
    }
  }
}

// ─────────────────────────────────────────────────────────────
int lese_adc_mittelwert(int samples = 20, int delay_ms = 10) {
  long summe = 0;
  for (int i = 0; i < samples; i++) {
    summe += analogRead(PIN_GAS);
    delay(delay_ms);
  }
  return (int)(summe / samples);
}

void kalibrierung_durchfuehren() {
  Serial.println("\n╔══════════════════════════════════════╗");
  Serial.println(  "║       KALIBRIERUNG GESTARTET         ║");
  Serial.println(  "║   Schrittweise Messung der Enden     ║");
  Serial.println(  "╚══════════════════════════════════════╝");

  int neuer_kein_gas = 0;
  int neuer_vollgas  = 0;

  while (true) {
    Serial.println("\n1) Trigger LOSLASSEN, dann 'y' senden");
    String antwort = warte_auf_eingabe();
    if (antwort != "y") {
      Serial.println("  Bitte 'y' eingeben.");
      continue;
    }
    neuer_kein_gas = lese_adc_mittelwert();
    Serial.printf("  Kein Gas ADC: %d\n", neuer_kein_gas);

    Serial.println("\n2) Trigger VOLL DRUECKEN, dann 'y' senden");
    antwort = warte_auf_eingabe();
    if (antwort != "y") {
      Serial.println("  Bitte 'y' eingeben.");
      continue;
    }
    neuer_vollgas = lese_adc_mittelwert();
    Serial.printf("  Vollgas ADC:  %d\n", neuer_vollgas);

    if (abs(neuer_kein_gas - neuer_vollgas) < 100) {
      Serial.println("\n⚠  Spreizung zu klein – Trigger wirklich bewegt?");
      Serial.println("   Kalibrierung verworfen, alte Werte bleiben.\n");
      return;
    }
    break;
  }

  // Ergebnis anzeigen + Bestätigung abfragen
  while (true) {
    Serial.println("\n┌──────────────────────────────────────┐");
    Serial.println(  "│       KALIBRIERUNG ABGESCHLOSSEN     │");
    Serial.println(  "├──────────────────────────────────────┤");
    Serial.printf(   "│  Kein Gas  → ADC: %-5d               │\n", neuer_kein_gas);
    Serial.printf(   "│  Vollgas   → ADC: %-5d               │\n", neuer_vollgas);
    Serial.printf(   "│  Spreizung:       %-5d counts         │\n", abs(neuer_kein_gas - neuer_vollgas));
    Serial.println(  "└──────────────────────────────────────┘");
    Serial.println("  Werte übernehmen? [y / n]");

    String antwort = warte_auf_eingabe();

    if (antwort == "y") {
      adc_kein_gas = neuer_kein_gas;
      adc_vollgas  = neuer_vollgas;
      Serial.println("✔  Werte übernommen!\n");
      break;
    } else if (antwort == "n") {
      Serial.println("↺  Wiederhole Kalibrierung...");
      kalibrierung_durchfuehren();  // rekursiv neu starten
      return;
    } else {
      Serial.println("  Bitte 'y' oder 'n' eingeben.");
    }
  }
}

// ─────────────────────────────────────────────────────────────
int lies_gas_prozent() {
  int roh    = analogRead(PIN_GAS);
  int spanne = adc_vollgas - adc_kein_gas;
  if (spanne == 0) return 0;

  // Robust gegen invertierte Kalibrierung: no-gas -> 0.0, full-gas -> 1.0
  float normiert = (float)(roh - adc_kein_gas) / (float)spanne;
  if (normiert < DEADZONE)             normiert = 0.0f;
  if (normiert > (1.0f - DEADZONE))    normiert = 1.0f;
  normiert = constrain(normiert, 0.0f, 1.0f);

  return (int)(normiert * 100.0f);
}

// ─────────────────────────────────────────────────────────────
bool lies_taster() {
  bool aktuell = digitalRead(PIN_TASTER);
  unsigned long jetzt = millis();

  if (aktuell != letzter_zustand && (jetzt - letzter_wechsel) > DEBOUNCE_MS) {
    letzter_zustand = aktuell;
    letzter_wechsel = jetzt;
    if (aktuell == LOW) return true;
  }
  return false;
}

// ─────────────────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);
  delay(500);

  analogReadResolution(12);
  analogSetAttenuation(ADC_11db);
  pinMode(PIN_TASTER, INPUT_PULLUP);

  Serial.println("\n════════════════════════════════════════");
  Serial.println("  Carrera Handregler bereit");
  Serial.printf( "  Standardwerte: Kein Gas=%d  Vollgas=%d\n", adc_kein_gas, adc_vollgas);
  Serial.printf( "  Taster-Pin: GPIO%d (INPUT_PULLUP)\n", PIN_TASTER);
  Serial.println("  Zum Kalibrieren: 'c' senden");
  Serial.println("════════════════════════════════════════\n");
}

// ─────────────────────────────────────────────────────────────
void loop() {
  // Serielle Eingabe prüfen
  if (Serial.available()) {
    String cmd = serial_lesen();
    if (cmd == "c") {
      kalibrierung_durchfuehren();
      Serial.println("  Weiter mit 'c' oder normal fahren.\n");
    }
  }

  static unsigned long letzter_print = 0;
  if (millis() - letzter_print > 200) {
    bool raw = digitalRead(PIN_TASTER);
    Serial.printf("Taster raw: %s\n", raw == LOW ? "LOW" : "HIGH");
    letzter_print = millis();
  }

  // Gas + Taster ausgeben
  int  gas         = lies_gas_prozent();
  bool spurwechsel = lies_taster();

  if (spurwechsel) Serial.println(">>> SPURWECHSEL! <<<");

  int balken = gas / 5;
  Serial.print("Gas: ");
  if (gas < 100) Serial.print(" ");
  if (gas < 10)  Serial.print(" ");
  Serial.print(gas);
  Serial.print("%  [");
  for (int i = 0; i < 20; i++) Serial.print(i < balken ? "█" : "░");
  Serial.println("]");

  delay(50);
}
